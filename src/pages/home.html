$naturalResults = $simpsonsPage->getNaturalResults();

foreach($naturalResults as $result){

    switch($result->getType()){

        case "classical":
            $resultTitle = $result->getTitle();
            $website     = $result->getWebsite();
            $url         = $result->getTargetUrl();
            break;


        case "pictures":
            foreach($result->getItems() as $imageResult){
                $url = $imageResult->getTargetUrl();
                $imageUrl = $imageResult->getImageUrl();

}
